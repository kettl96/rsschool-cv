function solution(numbers) { 
  const checkIfSame = numbers.reduce((acc, item) => {
    acc[item] = 1
    return acc
  }, {})
  if (Object.keys(checkIfSame).length > 1) {
    let largest = numbers[0]
    let smallest = numbers[0]
    numbers.forEach((item, index) => {
      if (largest < item) {
        largest = item
      }
      if (smallest > item) {
        smallest = item
      }
    })
    numbers.forEach((item, index) => {
      if (item === largest) {
        numbers.splice(index, 1, (largest - smallest))
      }
    })
    return solution(numbers)
  } else {
    const check = numbers.reduce((acc, item) => {
      acc += item
      return acc 
    }, 0)
    return check
  }
}
